#!/bin/bash

# --- 配置区域 ---
SSH_IP="10.174.176.201"
SSH_USER="g50060069"
SN="5JV0226127000574"
# 更新后的 D8 路径
D8_PATH="/home/gonglingwen/d8phoneBase/d8"
# 日志文件名
LOG="test2.log"

# --- 任务列表 (仅保留你指定的套件) ---
# 格式: "套件名|xflags"
tasks=(
    "safari|--test"
    "mjsunit|"
    "chakaracore|--module"
    "intl|--test"
    "intl-132|--test"
    "js-perf-test-132|--test"
)

# --- 执行逻辑 ---
# 清空旧日志
> $LOG

echo "==== 自动化测试 [精简版] 开始: $(date) ====" | tee -a $LOG
echo "使用 D8 路径: $D8_PATH" | tee -a $LOG

for i in "${!tasks[@]}"; do
    # 解析任务名和参数
    IFS='|' read -r suite flags <<< "${tasks[$i]}"
    count=$((i+1))

    echo "------------------------------------------------" | tee -a $LOG
    echo "进度: [$count/${#tasks[@]}] 正在执行: $suite" | tee -a $LOG

    # 构造命令 (注意这里使用了更新后的 D8_PATH)
    CMD="python run.py --suitename=$suite --mode=release --platform=phone --ssh-ip $SSH_IP --ssh-user $SSH_USER --sn $SN --d8=$D8_PATH --enable_cpltp"

    # 如果 flags 不为空，则加上 xflags 参数
    if [ ! -z "$flags" ]; then
        CMD="$CMD --xflags=\"$flags\""
    fi

    # 执行并实时记录输出到 test2.log
    eval $CMD 2>&1 | tee -a $LOG

    echo "任务 $suite 执行完毕。" | tee -a $LOG
done

echo "================================================" | tee -a $LOG
echo "所有精简测试已完成！日志保存在 $LOG" | tee -a $LOGgonglingwen@ubuntu:~/OpenHarmony_ArkTS_Test/JSVM/Function/V8_Lite$
